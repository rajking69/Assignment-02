#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n, q;
    cin >> n >> q;

    vector<long long> a(n);
    for (int i = 0; i < n; i++)
        cin >> a[i];
    sort(a.begin(), a.end());

    while (q--)
    {
        long long x;
        cin >> x;
        int left = 0;
        int right = n - 1;
        int index = -1;
        while (left <=right)
        {
            int mid = left + (right - left) / 2;
            if (a[mid] == x)
            {
                index = mid;
                break;
            }
            else if (a[mid] < x)
            {
                left = mid + 1;
            }
            else
            {
                right = mid - 1;
            }
        }
            if (index != -1)
                cout << "Found " << index << "\n";
            else
                cout << "Not Found\n";
        
    }
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++)
        cin >> a[i];

    // Bubble Sort Algorithm
     for(int i = 0; i < n - 1; i++)
    {
        for(int j = 0; j < n - i - 1; j++)
        {
            if(a[j] > a[j + 1])
            {
                int temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
        }
    }
    for (int i = 0; i < n; i++)
        cout << a[i] << " ";
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

int main()
{
    string a, b;
    cin >> a >> b;

    // if lengths are different, not anagram
    if (a.size() != b.size())
    {
        cout << "NO";
        return 0;
    }

    int freq[26] = {0};

    // count characters of first string
    for (int i = 0; i < a.size(); i++)
        freq[a[i] - 'a']++;

    // subtract characters of second string
    for (int i = 0; i < b.size(); i++)
        freq[b[i] - 'a']--;

    // check frequency
    for (int i = 0; i < 26; i++)
    {
        if (freq[i] != 0)
        {
            cout << "NO";
            return 0;
        }
    }

    cout << "YES";
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin >> s;

    string t = s;   // copy original
    int n = t.size();

    // manual reverse
    for (int i = 0; i < n / 2; i++)
    {
        char temp = t[i];
        t[i] = t[n - i - 1];
        t[n - i - 1] = temp;
    }

    // palindrome check
    if (s == t)
        cout << "YES";
    else
        cout << "NO";

    return 0;
}

#include <bits/stdc++.h>
using namespace std;
int main()
{
    string a, b;
    cin >> a >> b;

    // 1. Compare (lexicographical)
    if (a == b)
        cout << "Equal\n";
    else if (a < b)
        cout << a << "\n";
    else
        cout << b << "\n";

    // 2. Concatenation
    cout << a + b << "\n";

    // 3. Copy
    string c = a;
    cout << c << "\n";

    // 4. Reverse (without built-in function)
    for (int i = a.size() - 1; i >= 0; i--)
        cout << a[i];
    cout << "\n";

    return 0;
}

#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long a, b;
    cin >> a >> b;

    long long x = a, y = b;

    while (y != 0)
    {
        long long temp = x % y;
        x = y;
        y = temp;
    }
    long long gcd = x;
    long long lcm = (a / gcd) * b;
    cout << gcd << " " << lcm << endl;

    return 0;
}
